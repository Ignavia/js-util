    /**
 * An interface for every object that can be extended by plugins.
 *
 * @interface
 */
export default class Extensible {

    /**
     * Adds the given method to this object.
     *
     * @param {String|Symbol} name
     * The name of the method.
     *
     * @param {Function} f
     * The function to execute.
     *
     * @abstract
     */
    [addMethodSym](name, f) {}

    /**
     * Removes the method with the given name from this object.
     *
     * @param {String|Symbol} name
     * The name of the function to remove.
     *
     * @abstract
     */
    [removeMethodSym](name) {}

    /**
     * Adds the given plugins to this object by calling the register method on
     * each plugin and passing this object as an argument.
     *
     * @param {Plugin} plugins
     * The plugins to add.
     *
     * @abstract
     */
    [addPluginSym](plugin) {}

    /**
     * Removes the given plugins from this object by calling the register method
     * on each plugin and passing this object as an argument.
     *
     * @param {Plugin} plugins
     * The plugins to remove.
     *
     * @abstract
     */
    [removePluginSym](plugin) {}
}

/**
 * A symbol for the addMethod method in the Extensible interface.
 *
 * @type {Symbol}
 */
export const addMethodSym = Symbol("addMethod");

/**
 * A symbol for the removeMethod method in the Extensible interface.
 *
 * @type {Symbol}
 */
export const removeMethodSym = Symbol("removeMethod");

/**
 * A symbol for the addPlugin method in the Extensible interface.
 *
 * @type {Symbol}
 */
export const addPluginSym = Symbol("addPlugin");

/**
 * A symbol for the removePlugin method in the Extensible interface.
 *
 * @type {Symbol}
 */
export const removePluginSym = Symbol("removePlugin");

/**
 * A mixin for Extensible behavior.
 *
 * @type {Object}
 */
export const extensibleMixin = {
    addMethod(name, f, override = false) {
        if (!override && this[name]) {
            throw new Error(`A property ${name} exists already on ${this}.`);
        }
        this[name] = f;
        return this;
    },

    removeMethod(name) {
        if (!this[name]) {
            throw new Error(`A property ${name} does not exist on ${this}.`);
        }
        delete this[name];
        return this;
    },

    addPlugin(plugin) {
        plugin.register(this);
        return this;
    },

    removePlugin(plugin) {
        plugin.unregister(this);
        return this;
    }
};
